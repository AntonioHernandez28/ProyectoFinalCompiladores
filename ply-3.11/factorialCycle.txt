Program programForTesting;
VARS 
int: globalA; 

function int factorial(int: n)
VARS int: i, factorial; 
{
    i = 1; 
    factorial = 1; 
    while(i <= n) do {
        factorial = factorial * i; 
        i = i + 1; 
    }
    return(factorial); 
    
}

principal()
{
    VARS int: a, b; 
    a = 7; 
    b = factorial(a); 
    write(b);     
}
